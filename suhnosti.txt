class Waybill {
  label "Накладная ЭТРАН";

  // Идентификатор накладной
  attribute id {
    type uuid;
    key primary;
  }

  // Тип накладной
  attribute waybillType {
    type varchar(100);
    is nullable;
    sync Etran.waybill.waybill_type;
  }

  // Номер накладной
  attribute waybillNumber {
    type varchar(100);
    is nullable;
    sync Etran.waybill.waybill_number;
  }

  // ИД накладной (внешний идентификатор)
  attribute waybillIdentifier {
    type varchar(100);
    is nullable;
    sync Etran.waybill.waybill_identifier;
  }

  // Вид отправки
  attribute shipmentType {
    type varchar(100);
    is nullable;
    sync Etran.waybill.shipment_type;
  }

  // Скорость отправки
  attribute shipmentSpeed {
    type varchar(100);
    is nullable;
    sync Etran.waybill.shipment_speed;
  }

  // Тип бланка
  attribute formType {
    type varchar(100);
    is nullable;
    sync Etran.waybill.form_type;
  }

  // Дата истечения срока доставки
  attribute deliveryDeadline {
    type timestamp;
    is nullable;
    sync Etran.waybill.delivery_deadline;
  }

  // Дата создания накладной
  attribute waybillCreatedAt {
    type varchar(100);
    is nullable;
    sync Etran.waybill.waybill_created_at;
  }

  // Дата предъявления
  attribute submittedAt {
    type timestamp;
    is nullable;
    sync Etran.waybill.submitted_at;
  }

  // Дата согласования
  attribute approvedAt {
    type timestamp;
    is nullable;
    sync Etran.waybill.approved_at;
  }

  // Дата принятия приемосдатчиком
  attribute acceptedAt {
    type timestamp;
    is nullable;
    sync Etran.waybill.accepted_at;
  }

  // Дата принятия груза
  attribute cargoAcceptedAt {
    type timestamp;
    is nullable;
    sync Etran.waybill.cargo_accepted_at;
  }

  // Дата отправления
  attribute departureAt {
    type timestamp;
    is nullable;
    sync Etran.waybill.departure_at;
  }

  // Страна отправления
  attribute departureCountry {
    type varchar(255);
    is nullable;
    sync Etran.waybill.departure_country;
  }

  // Станция отправления
  attribute departureStation {
    type varchar(255);
    is nullable;
    sync Etran.waybill.departure_station;
  }

  // Код станции отправления
  attribute departureStationCode {
    type varchar(50);
    is nullable;
    sync Etran.waybill.departure_station_code;
  }

  // Страна назначения
  attribute destinationCountry {
    type varchar(255);
    is nullable;
    sync Etran.waybill.destination_country;
  }

  // Станция назначения
  attribute destinationStation {
    type varchar(255);
    is nullable;
    sync Etran.waybill.destination_station;
  }

  // Код станции назначения
  attribute destinationStationCode {
    type varchar(50);
    is nullable;
    sync Etran.waybill.destination_station_code;
  }

  // Грузоотправитель
  attribute shipperName {
    type varchar(500);
    is nullable;
    sync Etran.waybill.shipper_name;
  }

  // Адрес грузоотправителя
  attribute shipperAddress {
    type varchar(1000);
    is nullable;
    sync Etran.waybill.shipper_address;
  }

  // ТГНЛ грузоотправителя
  attribute shipperTgnl {
    type varchar(100);
    is nullable;
    sync Etran.waybill.shipper_tgnl;
  }

  // ОКПО грузоотправителя
  attribute shipperOkpo {
    type varchar(20);
    is nullable;
    sync Etran.waybill.shipper_okpo;
  }

  // Грузополучатель
  attribute consigneeName {
    type varchar(500);
    is nullable;
    sync Etran.waybill.consignee_name;
  }

  // Адрес грузополучателя
  attribute consigneeAddress {
    type varchar(1000);
    is nullable;
    sync Etran.waybill.consignee_address;
  }

  // ТГНЛ грузополучателя
  attribute consigneeTgnl {
    type varchar(100);
    is nullable;
    sync Etran.waybill.consignee_tgnl;
  }

  // ОКПО грузополучателя
  attribute consigneeOkpo {
    type varchar(20);
    is nullable;
    sync Etran.waybill.consignee_okpo;
  }

  // Плательщик
  attribute payer {
    type varchar(500);
    is nullable;
    sync Etran.waybill.payer;
  }

  // Код плательщика
  attribute payerCode {
    type varchar(50);
    is nullable;
    sync Etran.waybill.payer_code;
  }

  // Форма оплаты
  attribute paymentForm {
    type varchar(100);
    is nullable;
    sync Etran.waybill.payment_form;
  }

  // Место оплаты
  attribute paymentPlace {
    type varchar(255);
    is nullable;
    sync Etran.waybill.payment_place;
  }

  // Планируемый тип собственности вагонов
  attribute wagonOwnershipType {
    type varchar(100);
    is nullable;
    sync Etran.waybill.wagon_ownership_type;
  }

  // Планируемое число вагонов
  attribute plannedWagons {
    type varchar(50);
    is nullable;
    sync Etran.waybill.planned_wagons;
  }

  // Ответственное лицо
  attribute responsiblePerson {
    type varchar(255);
    is nullable;
    sync Etran.waybill.responsible_person;
  }

  /*===============================================

            НАВИГАЦИОННЫЕ АТРИБУТЫ И МЕТОДЫ
  
  ===============================================*/

  // Вагоны накладной
  method waybillRailwayCarriages {
    type WaybillRailwayCarriage[];
  }

  // Грузы накладной
  method waybillProducts {
    type WaybillProduct[];
  }

  // Специальные отметки
  method waybillSpecialMarks {
    type WaybillSpecialMark[];
  }

  // Тарифные отметки
  method waybillTariffMarks {
    type WaybillTariffMark[];
  }
}
